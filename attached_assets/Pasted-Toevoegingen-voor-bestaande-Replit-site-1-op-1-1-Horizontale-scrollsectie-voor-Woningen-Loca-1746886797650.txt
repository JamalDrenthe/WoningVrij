3. (Optioneel) Parallax met IntersectionObserver (voor lichte JS)

<section class="relative h-[100vh] overflow-hidden">
  <div class="absolute top-0 left-0 w-full h-full parallax-bg bg-[url('/img/bg.jpg')] bg-cover bg-center -z-10"></div>
  <div class="relative z-10 p-8 text-white">
    <h2 class="text-4xl font-bold">Ervaar wonen anders</h2>
  </div>
</section>

<script>
  const bg = document.querySelector('.parallax-bg');
  window.addEventListener('scroll', () => {
    const offset = window.scrollY * 0.5;
    bg.style.transform = `translateY(${offset}px)`;
  });
</script>
